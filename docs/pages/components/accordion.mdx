## Kode Eksempler

```jsx
import { Accordion, AccordionContent, AccordionItem, AccordionTrigger } from '@nostromo/ui-core'

export default function AccordionExample() {
  return (
    <Accordion type="single" collapsible>
      <AccordionItem value="item-1">
        <AccordionTrigger>Hvad er Nostromo UI?</AccordionTrigger>
        <AccordionContent>
          Nostromo UI er et omfattende UI bibliotek bygget med React, TypeScript og Tailwind CSS.
        </AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-2">
        <AccordionTrigger>Hvordan installerer jeg det?</AccordionTrigger>
        <AccordionContent>
          Du kan installere det med npm: <code>npm install @nostromo/ui-core</code>
        </AccordionContent>
      </AccordionItem>
    </Accordion>
  )
}
```

## Multiple Items

```jsx
import { Accordion, AccordionContent, AccordionItem, AccordionTrigger } from '@nostromo/ui-core'

export default function MultipleAccordion() {
  return (
    <Accordion type="multiple">
      <AccordionItem value="item-1">
        <AccordionTrigger>Sektion 1</AccordionTrigger>
        <AccordionContent>Indhold i sektion 1</AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-2">
        <AccordionTrigger>Sektion 2</AccordionTrigger>
        <AccordionContent>Indhold i sektion 2</AccordionContent>
      </AccordionItem>
    </Accordion>
  )
}
```

## Props

| Prop | Type | Default | Beskrivelse |
|------|------|---------|-------------|
| `type` | `"single" \| "multiple"` | `"single"` | Accordion type |
| `collapsible` | `boolean` | `false` | Om items kan lukkes |
