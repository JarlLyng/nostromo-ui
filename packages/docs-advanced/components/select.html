Select

Dropdown select components with search, multi-select, and keyboard navigation. Perfect for forms and data selection.

Installation

import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@nostromo/ui-core'

Basic Usage

The Select component provides a dropdown interface for selecting options.

import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@nostromo/ui-core'

export default function BasicSelect() {
  return (
    <Select>
      <SelectTrigger>
        <SelectValue placeholder="Select an option" />
      </SelectTrigger>
      <SelectContent>
        <SelectItem value="option1">Option 1</SelectItem>
        <SelectItem value="option2">Option 2</SelectItem>
        <SelectItem value="option3">Option 3</SelectItem>
      </SelectContent>
    </Select>
  )
}
