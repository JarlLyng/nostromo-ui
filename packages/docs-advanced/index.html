Quick Start

Get started with Nostromo UI in minutes. Install the package and start building beautiful interfaces.

# Install with pnpm
pnpm add @nostromo/ui-core

# Or with npm
npm install @nostromo/ui-core

# Or with yarn
yarn add @nostromo/ui-core
import { Button } from '@nostromo/ui-core'

export default function App() {
  return (
    <div>
      <Button variant="primary">Launch Mission</Button>
      <Button variant="secondary">Abort Mission</Button>
    </div>
  )
}
